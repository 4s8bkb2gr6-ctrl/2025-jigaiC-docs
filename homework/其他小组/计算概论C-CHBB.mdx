        # 36 多少种取法 Solution by Xuzheng Chen
        T = int(input())
        def func(m, n, s):
            if s == 0 and n != 0:
                return 0
            if s == 0 and n == 0:
                return 1
            if n == 1 and m >= s:
                return 1
            if n == 1 and m < s:
                return 0
            if m < n:
                return 0
            if m == 0 and not (n == 0 and s == 0):
                return 0
            if s < m:
                return func(s, n, s)
            else:
                return (func(m-1, n-1, s-m) + func(m-1, n, s))
